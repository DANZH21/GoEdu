package main

import (
	"fmt"

	"github.com/k0kubun/pp"
)

type phone struct {
	Name string
	GB   float64
}

func main() {
	IPhone := NewPhone("азазазпенис", 5024050.00) // передал не валид

	pp.Println(IPhone)

	IPhone = NewPhone("азазазпенис", 541.00) // передал  валид

	pp.Println(IPhone)

}

func NewPhone(Name string, GB float64) phone {
	if Name == "" {
		fmt.Println("Наканчал, так наканчал")
		return phone{}
	} // проверка, не пустой ли нам дали
	if GB < 32 || GB > 2048 {
		fmt.Println("Хуйня брат")
		return phone{}
	}

	fmt.Println("Норм, давай сифончик")
	return phone{
		Name: Name,
		GB:   GB,
	}
}
